long_df <- df %>%
    pivot_longer(
        cols = starts_with("Biofilm formation"),
        names_to = "Species",
        values_to = "Expression"
    ) %>%
    mutate(
        Species = recode(Species,
                         "Biofilm formation - P.aeruginosa" = "P. aeruginosa",
                         "Biofilm formation - V.cholerae"   = "V. cholerae",
                         "Biofilm formation - E.coli"       = "E. coli"
        ),
        Species = factor(Species, levels = c("P. aeruginosa", "V. cholerae", "E. coli")),
        Location = factor(Location, levels = c("Cooler", "Hotbox"))
    )


pvals <- long_df %>%
    group_by(Species) %>%
    wilcox_test(Expression ~ Location, detailed = TRUE) %>%
    mutate(
        p_label = rstatix::p_format(p, digits = 3),
        group1 = "Cooler", group2 = "Hotbox"
    ) %>%
    left_join(
        long_df %>%
            group_by(Species) %>%
            summarise(y.position = max(Expression) * 1.08),
        by = "Species"
    )


italic_labeller <- as_labeller(
    c("P. aeruginosa" = "italic('P. aeruginosa')",
      "V. cholerae"   = "italic('V. cholerae')",
      "E. coli"       = "italic('E. coli')"),
    default = label_parsed
)


loc_colors <- c("Cooler" = "#1f78b4", "Hotbox" = "#e31a1c")

p <- ggplot(long_df, aes(x = Location, y = Expression, fill = Location)) +
    geom_boxplot(width = 0.65, alpha = 0.9, color = "black", outlier.shape = 21) +
    geom_jitter(width = 0.1, size = 2, alpha = 0.7) +
    scale_fill_manual(values = loc_colors) +
    facet_wrap(~ Species, ncol = 3, scales = "free_y",
               labeller = italic_labeller) +        # <-- italic strip labels
    labs(title = "KEGG - Biofilm Formation (Cooler vs Hotbox)",
         x = NULL, y = "Expression (TPM)") +
    theme_classic(base_size = 14) +
    theme(
        legend.position  = "none",
        strip.text       = element_text(size = 13, face = "italic"),  # <-- italic font
        plot.title       = element_text(hjust = 0.5, face = "bold", size = 16)
    ) +
    ggpubr::stat_pvalue_manual(
        data         = pvals,
        label        = "p_label",
        xmin         = "group1", xmax = "group2",
        y.position   = "y.position",
        tip.length   = 0.01, bracket.size = 0.5,
        inherit.aes  = FALSE
    )

print(p)
ggsave("biofilm_boxplot.png", p, width = 9, height = 5.5, dpi = 300)
